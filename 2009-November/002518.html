import cairo

mapfile = 'world_styles.xml'
map_output = 'mapfile.pdf'

mapnik.load_map(mapnik_map, mapfile)

surface = cairo.PDFSurface(file.name, mapnik_map.width, mapnik_map.height)
mapnik.render(mapnik_map, surface)
surface.finish()
